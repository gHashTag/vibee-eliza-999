### –®–∞–≥ 4: –°–æ–∑–¥–∞–Ω–∏–µ –¥–µ–π—Å—Ç–≤–∏—è (Action)

```typescript
// src/agents/{agent-name}/actions/{action-name}.ts
import { TaskEither, tryCatchAsync, right, left } from '@/core/functional/utils/result'
import { pipe, chain, map, tap } from '@/core/functional/utils/composition'
import { validate } from '@/core/validation/validate'
import { withLogging } from '@/core/sentry/withLogging'
import { withMetrics } from '../utils/metrics'
import { withCache } from '../utils/cache'
import { checkRateLimit } from '../utils/rateLimit'
import { retryWithBackoff } from '../utils/retry'
import { withTimeout } from '../utils/timeout'
import { categorizeError } from '../utils/errorHandler'
import { {ActionName}InputSchema, {ActionName}OutputSchema } from '../schemas'
import { db } from '@/core/drizzle/client'
import { operations, assets, users, balances } from '@/core/drizzle/schema'
import { eq, and } from 'drizzle-orm'
import { sql } from 'drizzle-orm'
import { AvatarBrain } from '@/core/types'
import { logger } from '@/core/logger'

/**
 * –î–µ–π—Å—Ç–≤–∏–µ: {ActionName}
 * 
 * –û–ø–∏—Å–∞–Ω–∏–µ: {–ß—Ç–æ –¥–µ–ª–∞–µ—Ç —ç—Ç–æ –¥–µ–π—Å—Ç–≤–∏–µ}
 */
export const {actionName}Action = {
  name: '{action-name}',
  description: '{–û–ø–∏—Å–∞–Ω–∏–µ –¥–µ–π—Å—Ç–≤–∏—è}',
  
  /**
   * –í–∞–ª–∏–¥–∞—Ü–∏—è –≤—Ö–æ–¥–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö –∏ –∫–æ–Ω—Ç–µ–∫—Å—Ç–∞
   */
  validate: (
    brain: AvatarBrain,
    context: unknown
  ): TaskEither<Error, boolean> => {
    return pipe(
      // 1. –í–∞–ª–∏–¥–∞—Ü–∏—è –≤—Ö–æ–¥–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö —á–µ—Ä–µ–∑ Zod
      validate({ActionName}InputSchema, context),
      chain(validatedInput => {
        // 2. –ü—Ä–æ–≤–µ—Ä–∫–∞ –±–∞–ª–∞–Ω—Å–∞ (–µ—Å–ª–∏ –Ω—É–∂–Ω–æ)
        const cost = 10 // –°—Ç–æ–∏–º–æ—Å—Ç—å –æ–ø–µ—Ä–∞—Ü–∏–∏
        const currentBalance = (brain.balance?.currencies as any)?.stars || 0
        
        if (currentBalance < cost) {
          return left(new Error(`–ù–µ–¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ –±–∞–ª–∞–Ω—Å–∞. –¢—Ä–µ–±—É–µ—Ç—Å—è: ${cost}‚≠ê`))
        }
        
        // 3. –ü—Ä–æ–≤–µ—Ä–∫–∞ –ø—Ä–∞–≤ –¥–æ—Å—Ç—É–ø–∞ (–µ—Å–ª–∏ –Ω—É–∂–Ω–æ)
        // if (!hasPermission(brain, 'action-name')) {
        //   return left(new Error('–ù–µ—Ç –ø—Ä–∞–≤ –¥–æ—Å—Ç—É–ø–∞'))
        // }
        
        return right(true)
      })
    )
  },
  
  /**
   * –û–±—Ä–∞–±–æ—Ç—á–∏–∫ –¥–µ–π—Å—Ç–≤–∏—è —Å –ø–æ–ª–Ω–æ–π –æ–±—Ä–∞–±–æ—Ç–∫–æ–π –æ—à–∏–±–æ–∫ –∏ –º–µ—Ç—Ä–∏–∫–∞–º–∏
   */
  handler: (
    brain: AvatarBrain,
    context: unknown
  ): TaskEither<Error, {ActionName}Output> => {
    // –û–±–µ—Ä—Ç–∫–∞ —Å –ª–æ–≥–∏—Ä–æ–≤–∞–Ω–∏–µ–º, –º–µ—Ç—Ä–∏–∫–∞–º–∏ –∏ timeout
    return pipe(
      // 1. –ü—Ä–æ–≤–µ—Ä–∫–∞ rate limit
      checkRateLimit({
        maxRequests: 10,
        windowMs: 60000, // 1 –º–∏–Ω—É—Ç–∞
        key: `agent:${brain.id}:{action-name}`,
      }),
      chain(() =>
        pipe(
          // 2. –í–∞–ª–∏–¥–∞—Ü–∏—è –≤—Ö–æ–¥–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö
          validate({ActionName}InputSchema, context),
          chain(validatedInput =>
            pipe(
              // 3. Timeout –¥–ª—è –æ–ø–µ—Ä–∞—Ü–∏–∏ (30 —Å–µ–∫—É–Ω–¥)
              withTimeout(
                pipe(
                  // 4. –í—ã–ø–æ–ª–Ω–µ–Ω–∏–µ –æ—Å–Ω–æ–≤–Ω–æ–π –ª–æ–≥–∏–∫–∏ —Å retry
                  executeActionLogic(validatedInput, brain),
                  chain(result =>
                    pipe(
                      // 5. –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞ –≤ –ë–î (—Ç—Ä–∞–Ω–∑–∞–∫—Ü–∏—è)
                      saveResultToDB(result, brain),
                      chain(() =>
                        pipe(
                          // 6. –°–ø–∏—Å–∞–Ω–∏–µ –±–∞–ª–∞–Ω—Å–∞ –∞—Ç–æ–º–∞—Ä–Ω–æ (–µ—Å–ª–∏ –Ω—É–∂–Ω–æ)
                          deductBalance(brain, cost),
                          map(() => ({
                            success: true,
                            data: result,
                            timestamp: new Date(),
                          }))
                        )
                      )
                    )
                  )
                ),
                30000 // 30 —Å–µ–∫—É–Ω–¥ timeout
              ),
              // 7. –û–±—Ä–∞–±–æ—Ç–∫–∞ –æ—à–∏–±–æ–∫ —Å –∫–∞—Ç–µ–≥–æ—Ä–∏–∑–∞—Ü–∏–µ–π
              chain(
                (result) => right(result),
                (error) => {
                  const categorized = categorizeError(error)
                  
                  // –õ–æ–≥–∏—Ä—É–µ–º –æ—à–∏–±–∫—É
                  logger.error('Action handler error', {
                    agent: '{AGENT-NAME}',
                    action: '{action-name}',
                    category: categorized.category,
                    retryable: categorized.retryable,
                    error: categorized.technicalMessage,
                    userId: brain.id,
                  })
                  
                  // –í–æ–∑–≤—Ä–∞—â–∞–µ–º –∫–∞—Ç–µ–≥–æ—Ä–∏–∑–∏—Ä–æ–≤–∞–Ω–Ω—É—é –æ—à–∏–±–∫—É
                  return left(categorized)
                }
              )
            )
          )
        )
      )
    )
  },
}

/**
 * –û—Å–Ω–æ–≤–Ω–∞—è –ª–æ–≥–∏–∫–∞ –¥–µ–π—Å—Ç–≤–∏—è —Å retry –∏ –æ–±—Ä–∞–±–æ—Ç–∫–æ–π –æ—à–∏–±–æ–∫
 */
const executeActionLogic = (
  input: {ActionName}Input,
  brain: AvatarBrain
): TaskEither<Error, unknown> => {
  return pipe(
    // 1. –í–∞–ª–∏–¥–∞—Ü–∏—è –≤—Ö–æ–¥–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö
    validateInput(input),
    chain(validated =>
      pipe(
        // 2. –í—ã–ø–æ–ª–Ω–µ–Ω–∏–µ —Å retry –ª–æ–≥–∏–∫–æ–π
        retryWithBackoff(
          () => performAction(validated, brain),
          {
            maxRetries: 3,
            delays: [1000, 2000, 4000], // Exponential backoff
            retryableErrors: ['TIMEOUT', 'NETWORK_ERROR', 'RATE_LIMIT'],
          }
        ),
        chain(result =>
          pipe(
            // 3. –í–∞–ª–∏–¥–∞—Ü–∏—è —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞
            validateResult(result),
            map(validatedResult => ({
              ...validatedResult,
              metadata: {
                executedAt: new Date(),
                userId: brain.id,
                agent: '{AGENT-NAME}',
              },
            }))
          )
        )
      )
    )
  )
}

/**
 * –í—ã–ø–æ–ª–Ω–µ–Ω–∏–µ –¥–µ–π—Å—Ç–≤–∏—è —Å –æ–±—Ä–∞–±–æ—Ç–∫–æ–π –æ—à–∏–±–æ–∫
 */
const performAction = (
  input: {ActionName}Input,
  brain: AvatarBrain
): TaskEither<Error, unknown> => {
  return tryCatchAsync(
    async () => {
      // –†–µ–∞–ª–∏–∑—É–π –æ—Å–Ω–æ–≤–Ω—É—é –ª–æ–≥–∏–∫—É –∑–¥–µ—Å—å
      // –ò—Å–ø–æ–ª—å–∑—É–π —Ñ—É–Ω–∫—Ü–∏–æ–Ω–∞–ª—å–Ω—ã–π —Å—Ç–∏–ª—å (pipe, chain, map)
      
      // –ü—Ä–∏–º–µ—Ä –≤—ã–∑–æ–≤–∞ –≤–Ω–µ—à–Ω–µ–≥–æ API:
      // const response = await callExternalAPI(input)
      // return response
      
      // –ü—Ä–∏–º–µ—Ä —Ä–∞–±–æ—Ç—ã —Å –ë–î:
      // const data = await db.select().from(table).where(...)
      // return data
      
      return {
        result: 'Success',
        data: input,
      }
    },
    (error) => {
      // –ö–∞—Ç–µ–≥–æ—Ä–∏–∑–∞—Ü–∏—è –æ—à–∏–±–æ–∫ –¥–ª—è –ø—Ä–∞–≤–∏–ª—å–Ω–æ–π –æ–±—Ä–∞–±–æ—Ç–∫–∏
      if (error instanceof Error) {
        // –°–µ—Ç–µ–≤—ã–µ –æ—à–∏–±–∫–∏ - –º–æ–∂–Ω–æ –ø–æ–≤—Ç–æ—Ä–∏—Ç—å
        if (error.message.includes('timeout') || error.message.includes('network')) {
          return new Error(`NETWORK_ERROR: ${error.message}`)
        }
        
        // –û—à–∏–±–∫–∏ –≤–∞–ª–∏–¥–∞—Ü–∏–∏ - –Ω–µ –ø–æ–≤—Ç–æ—Ä—è–µ–º
        if (error.message.includes('validation') || error.message.includes('invalid')) {
          return new Error(`VALIDATION_ERROR: ${error.message}`)
        }
        
        // –û—à–∏–±–∫–∏ –∞–≤—Ç–æ—Ä–∏–∑–∞—Ü–∏–∏ - –Ω–µ –ø–æ–≤—Ç–æ—Ä—è–µ–º
        if (error.message.includes('unauthorized') || error.message.includes('403')) {
          return new Error(`AUTH_ERROR: ${error.message}`)
        }
      }
      
      return error instanceof Error ? error : new Error(String(error))
    }
  )
}

/**
 * –í–∞–ª–∏–¥–∞—Ü–∏—è –≤—Ö–æ–¥–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö
 */
const validateInput = (
  input: {ActionName}Input
): TaskEither<Error, {ActionName}Input> => {
  return validate({ActionName}InputSchema, input)
}

/**
 * –í–∞–ª–∏–¥–∞—Ü–∏—è —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞
 */
const validateResult = (
  result: unknown
): TaskEither<Error, {ActionName}Output> => {
  return validate({ActionName}OutputSchema, result)
}

/**
 * –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞ –≤ –ë–î —á–µ—Ä–µ–∑ Drizzle ORM —Å —Ç—Ä–∞–Ω–∑–∞–∫—Ü–∏–µ–π
 */
const saveResultToDB = (
  result: unknown,
  brain: AvatarBrain
): TaskEither<Error, void> => {
  return tryCatchAsync(
    async () => {
      // –ò—Å–ø–æ–ª—å–∑—É–π —Ç—Ä–∞–Ω–∑–∞–∫—Ü–∏—é Drizzle –¥–ª—è –∞—Ç–æ–º–∞—Ä–Ω–æ—Å—Ç–∏
      await db.transaction(async (tx) => {
        // 1. –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ –æ–ø–µ—Ä–∞—Ü–∏–∏
        const [operation] = await tx
          .insert(operations)
          .values({
            user_id: brain.id,
            type: '{operation-type}',
            service_type: '{service-type}',
            status: 'completed',
            cost: {
              stars: cost,
            },
            result: result as any,
            metadata: {
              agent: '{AGENT-NAME}',
              action: '{action-name}',
              timestamp: new Date(),
            },
          })
          .returning()
        
        // 2. –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ –∞—Å—Å–µ—Ç–∞ (–µ—Å–ª–∏ –µ—Å—Ç—å)
        if (result && typeof result === 'object' && 'asset' in result) {
          await tx.insert(assets).values({
            user_id: brain.id,
            type: '{asset-type}',
            url: (result as any).asset.url,
            metadata: {
              operation_id: operation.id,
              agent: '{AGENT-NAME}',
            },
          })
        }
        
        // 3. –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∏ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
        await tx
          .update(users)
          .set({
            updated_at: new Date(),
          })
          .where(eq(users.id, brain.id))
      })
    },
    (error) => {
      // –õ–æ–≥–∏—Ä—É–µ–º –æ—à–∏–±–∫—É –ë–î
      logger.error('Failed to save result to DB', {
        error: error instanceof Error ? error.message : String(error),
        userId: brain.id,
        agent: '{AGENT-NAME}',
      })
      return error instanceof Error ? error : new Error(String(error))
    }
  )
}

/**
 * –°–ø–∏—Å–∞–Ω–∏–µ –±–∞–ª–∞–Ω—Å–∞ –∞—Ç–æ–º–∞—Ä–Ω–æ —Å –ø—Ä–æ–≤–µ—Ä–∫–æ–π –∏ –±–ª–æ–∫–∏—Ä–æ–≤–∫–æ–π
 */
const deductBalance = (
  brain: AvatarBrain,
  cost: number
): TaskEither<Error, void> => {
  return tryCatchAsync(
    async () => {
      // –ò—Å–ø–æ–ª—å–∑—É–π —Ç—Ä–∞–Ω–∑–∞–∫—Ü–∏—é Drizzle –¥–ª—è –∞—Ç–æ–º–∞—Ä–Ω–æ—Å—Ç–∏
      await db.transaction(async (tx) => {
        // 1. –ë–ª–æ–∫–∏—Ä—É–µ–º —Å—Ç—Ä–æ–∫—É –±–∞–ª–∞–Ω—Å–∞ –¥–ª—è –∞—Ç–æ–º–∞—Ä–Ω–æ—Å—Ç–∏ (SELECT FOR UPDATE)
        const [balance] = await tx
          .select()
          .from(balances)
          .where(eq(balances.user_id, brain.id))
          .for('update') // –ë–ª–æ–∫–∏—Ä–æ–≤–∫–∞ —Å—Ç—Ä–æ–∫–∏
          .limit(1)
        
        if (!balance) {
          throw new Error('Balance not found')
        }
        
        // 2. –ü—Ä–æ–≤–µ—Ä—è–µ–º –¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ—Å—Ç—å –±–∞–ª–∞–Ω—Å–∞
        const currentStars = (balance.currencies as any)?.stars || 0
        if (currentStars < cost) {
          throw new Error(`Insufficient balance. Current: ${currentStars}‚≠ê, Required: ${cost}‚≠ê`)
        }
        
        // 3. –ê—Ç–æ–º–∞—Ä–Ω–æ–µ —Å–ø–∏—Å–∞–Ω–∏–µ –±–∞–ª–∞–Ω—Å–∞ —á–µ—Ä–µ–∑ SQL —Ñ—É–Ω–∫—Ü–∏—é
        await tx.execute(sql`
          UPDATE balances
          SET currencies = jsonb_set(
            currencies,
            '{stars}',
            ((currencies->>'stars')::int - ${cost})::text::jsonb
          ),
          updated_at = NOW()
          WHERE user_id = ${brain.id}
        `)
        
        // 4. –°–æ–∑–¥–∞–µ–º –∑–∞–ø–∏—Å—å –æ–± –æ–ø–µ—Ä–∞—Ü–∏–∏ –¥–ª—è –∞—É–¥–∏—Ç–∞
        await tx.insert(operations).values({
          user_id: brain.id,
          type: 'BALANCE_DEDUCTION',
          service_type: '{service-type}',
          status: 'completed',
          cost: {
            stars: cost,
          },
          metadata: {
            agent: '{AGENT-NAME}',
            action: '{action-name}',
            previous_balance: currentStars,
            new_balance: currentStars - cost,
          },
        })
      })
    },
    (error) => {
```
