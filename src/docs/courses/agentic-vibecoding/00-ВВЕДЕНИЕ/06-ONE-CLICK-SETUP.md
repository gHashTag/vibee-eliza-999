get_version() {
  local tool=$1
  case $tool in
    git)   git --version 2>/dev/null | awk '{print $3}' ;;
    node)  node --version 2>/dev/null ;;

should_skip() {

      return 0

  done
  return 1
}

install_git() {

      return 0
    fi
  fi

  log INFO "–£—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º $tool..."

  local os=$(detect_os)
  if [[ "$DRY_RUN" == true ]]; then
    log INFO "[DRY RUN] –£—Å—Ç–∞–Ω–æ–≤–∫–∞ $tool –Ω–∞ $os"
  else
    case $os in

        brew install git || { log ERROR "–ù–µ —É–¥–∞–ª–æ—Å—å —É—Å—Ç–∞–Ω–æ–≤–∏—Ç—å $tool"; return 1; }
        ;;
      linux)

        { log ERROR "–ù–µ —É–¥–∞–ª–æ—Å—å —É—Å—Ç–∞–Ω–æ–≤–∏—Ç—å $tool"; return 1; }
        ;;
      *)
        log ERROR "–ù–µ–ø–æ–¥–¥–µ—Ä–∂–∏–≤–∞–µ–º–∞—è –û–°: $os"
        return 1
        ;;
    esac
  fi

  log SUCCESS "$tool —É—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω: $(get_version $tool)"
}
